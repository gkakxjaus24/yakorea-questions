    // ========== SVG ==========
    const ICON_CHAT = '<svg viewBox="0 0 24 24"><path d="M20 2H4c-1.1 0-2 .9-2 2v18l4-4h14c1.1 0 2-.9 2-2V4c0-1.1-.9-2-2-2zm0 14H5.17L4 17.17V4h16v12z"/><path d="M7 9h2v2H7zm4 0h2v2h-2zm4 0h2v2h-2z"/></svg>';
    const ICON_CLOSE = '✖';
    const ICON_SEND = '<svg viewBox="0 0 24 24"><path d="M2.01 21L23 12 2.01 3 2 10l15 2-15 2z"/></svg>';

    // ========== Widget Class ==========
    class YakoreaChat {
        constructor() {
            this.isOpen = false;
            this.language = 'ko';
            this.sessionId = this._getSessionId();
            this.roomId = null;
            this.socket = null;
            this.managerStatus = 'none';
            this.unreadCount = 0;
            this.host = null;
            this.shadow = null;
            this.elements = {};

            this._loadSocketIO()
                .then(() => this._createWidget())
                .then(() => this._connectSocket())
                .then(() => this._restoreSession())
                .catch(err => console.error('Yakorea Chat init error:', err));
        }

        _getSessionId() {
            let id = localStorage.getItem('yakorea_chat_session');
            if (!id) {
                id = 'guest_' + Date.now() + '_' + Math.random().toString(36).substring(2, 9);
                localStorage.setItem('yakorea_chat_session', id);
            }
            return id;
        }

        _loadSocketIO() {
            return new Promise((resolve, reject) => {
                if (window.io) { resolve(); return; }
                const script = document.createElement('script');
                script.src = SERVER_URL + '/socket.io/socket.io.js';
                script.onload = resolve;
                script.onerror = () => reject(new Error('Socket.IO load failed'));
                document.head.appendChild(script);
            });
        }

        _createWidget() {
            const t = I18N[this.language];
            this.host = document.createElement('div');
            this.host.id = 'yakorea-chat-widget';
            this.shadow = this.host.attachShadow({ mode: 'open' });

            const style = document.createElement('style');
            style.textContent = WIDGET_CSS;
            this.shadow.appendChild(style);

            const container = document.createElement('div');
            container.innerHTML = `
        <button class="chat-toggle-btn" id="chatToggle" aria-label="Chat">
          \${ICON_CHAT}
          <span class="chat-badge" id="chatBadge"></span>
        </button>
        <div class="chat-window" id="chatWindow">
          <div class="chat-header">
            <div class="chat-header-left">
              <div class="chat-header-title" id="headerTitle">\${t.title}</div>
              <div class="chat-header-status">
                <span class="status-dot" id="statusDot"></span>
                <span id="statusText">\${t.connecting}</span>
              </div>
            </div>
            <div class="chat-header-actions">
              <button id="closeBtn" aria-label="Close">\${ICON_CLOSE}</button>
            </div>
          </div>
          <div class="language-bar" id="langBar">
            <button data-lang="ko" class="active">한국어</button>
            <button data-lang="en">English</button>
            <button data-lang="zh">中文</button>
            <button data-lang="ja">日本語</button>
          </div>
          <div class="chat-messages" id="chatMessages"></div>
          <button class="manager-connect-btn" id="managerBtn">\${t.manager_btn}</button>
          <div class="chat-input-area">
            <textarea id="chatInput" rows="1" placeholder="\${t.placeholder}"></textarea>
            <button class="chat-send-btn" id="sendBtn" disabled>\${ICON_SEND}</button>
          </div>
          <div class="chat-footer"><span>Yakorea Chat</span></div>
        </div>
      `;

            this.shadow.appendChild(container);
            document.body.appendChild(this.host);

            this.elements = {
                toggle: this.shadow.getElementById('chatToggle'),
                badge: this.shadow.getElementById('chatBadge'),
                window: this.shadow.getElementById('chatWindow'),
                headerTitle: this.shadow.getElementById('headerTitle'),
                statusDot: this.shadow.getElementById('statusDot'),
                statusText: this.shadow.getElementById('statusText'),
                langBar: this.shadow.getElementById('langBar'),
                messages: this.shadow.getElementById('chatMessages'),
                managerBtn: this.shadow.getElementById('managerBtn'),
                input: this.shadow.getElementById('chatInput'),
                sendBtn: this.shadow.getElementById('sendBtn'),
                closeBtn: this.shadow.getElementById('closeBtn')
            };

            this._bindEvents();
        }

        _bindEvents() {
            const el = this.elements;
            el.toggle.addEventListener('click', () => this._toggleChat());
            el.closeBtn.addEventListener('click', () => this._toggleChat(false));
            el.langBar.querySelectorAll('button').forEach(btn => {
                btn.addEventListener('click', () => this._changeLanguage(btn.dataset.lang));
            });
            el.input.addEventListener('input', () => {
                el.sendBtn.disabled = el.input.value.trim() === '';
                el.input.style.height = 'auto';
                el.input.style.height = Math.min(el.input.scrollHeight, 100) + 'px';
            });
            el.sendBtn.addEventListener('click', () => this._sendMessage());
            el.input.addEventListener('keydown', (e) => {
                if (e.key === 'Enter' && !e.shiftKey) {
                    e.preventDefault();
                    if (el.input.value.trim()) this._sendMessage();
                }
            });
            el.managerBtn.addEventListener('click', () => this._requestManager());
        }

        _connectSocket() {
            this.socket = window.io(SERVER_URL, { transports: ['websocket', 'polling'] });
            this.socket.on('connect', () => {
                const t = I18N[this.language];
                this.elements.statusDot.classList.remove('offline');
                this.elements.statusText.textContent = t.connected;
                if (this.roomId) {
                    this.socket.emit('guest:join', { roomId: this.roomId, sessionId: this.sessionId });
                }
            });
            this.socket.on('disconnect', () => {
                const t = I18N[this.language];
                this.elements.statusDot.classList.add('offline');
                this.elements.statusText.textContent = t.disconnected;
            });
            this.socket.on('new_message', (data) => {
                const msg = data.message;
                if (msg.sender_type === 'guest') return;
                this._addMessageToUI(msg);
                if (msg.sender_type === 'manager') this._updateManagerStatus('connected');
                if (!this.isOpen) { this.unreadCount++; this._updateBadge(); }
            });
        }

        async _restoreSession() {
            try {
                const savedRoomId = localStorage.getItem('yakorea_chat_room');
                const savedLang = localStorage.getItem('yakorea_chat_lang');
                if (savedLang && I18N[savedLang]) this._changeLanguage(savedLang);
                if (!savedRoomId) return;
                const roomRes = await fetch(SERVER_URL + '/api/chat/rooms/' + savedRoomId);
                if (!roomRes.ok) { localStorage.removeItem('yakorea_chat_room'); return; }
                const { room } = await roomRes.json();
                if (room.status === 'closed') { localStorage.removeItem('yakorea_chat_room'); return; }
                this.roomId = savedRoomId;
                const msgRes = await fetch(SERVER_URL + '/api/chat/rooms/' + savedRoomId + '/messages');
                if (msgRes.ok) {
                    const { messages } = await msgRes.json();
                    messages.forEach(msg => this._addMessageToUI(msg));
                }
                if (this.socket && this.socket.connected) {
                    this.socket.emit('guest:join', { roomId: this.roomId, sessionId: this.sessionId });
                }
                if (room.is_manager_connected) this._updateManagerStatus('connected');
                else if (room.status === 'waiting_manager') this._updateManagerStatus('requested');
            } catch (error) {
                console.error('Session restore error:', error);
            }
        }

        _toggleChat(forceState) {
            this.isOpen = forceState !== undefined ? forceState : !this.isOpen;
            this.elements.window.classList.toggle('open', this.isOpen);
            if (this.isOpen) {
                this.unreadCount = 0;
                this._updateBadge();
                setTimeout(() => this.elements.input.focus(), 300);
                this._scrollToBottom();
            }
        }

        _changeLanguage(lang) {
            if (!I18N[lang]) return;
            this.language = lang;
            localStorage.setItem('yakorea_chat_lang', lang);
            const t = I18N[lang];
            this.elements.headerTitle.textContent = t.title;
            this.elements.input.placeholder = t.placeholder;
            if (this.socket && this.socket.connected) {
                this.elements.statusText.textContent = t.connected;
            } else {
                this.elements.statusText.textContent = t.disconnected;
            }
            if (this.managerStatus === 'connected') this.elements.managerBtn.textContent = t.manager_connected;
            else if (this.managerStatus === 'requested') this.elements.managerBtn.textContent = t.manager_requested;
            else this.elements.managerBtn.textContent = t.manager_btn;
            this.elements.langBar.querySelectorAll('button').forEach(btn => {
                btn.classList.toggle('active', btn.dataset.lang === lang);
            });
        }

        async _sendMessage() {
            const content = this.elements.input.value.trim();
            if (!content) return;
            this.elements.input.value = '';
            this.elements.input.style.height = 'auto';
            this.elements.sendBtn.disabled = true;
            try {
                if (!this.roomId) {
                    const res = await fetch(SERVER_URL + '/api/chat/rooms', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ guestSessionId: this.sessionId, language: this.language })
                    });
                    if (!res.ok) throw new Error('Room creation failed');
                    const { room } = await res.json();
                    this.roomId = room.id;
                    localStorage.setItem('yakorea_chat_room', this.roomId);
                    this.socket.emit('guest:join', { roomId: this.roomId, sessionId: this.sessionId });
                }
                const tempMsg = {
                    id: 'temp_' + Date.now(),
                    sender_type: 'guest',
                    content: content,
                    created_at: new Date().toISOString()
                };
                this._addMessageToUI(tempMsg);
                this.socket.emit('guest:send_message', { roomId: this.roomId, content: content });
            } catch (error) {
                console.error('Message send error:', error);
            }
        }

        _addMessageToUI(msg) {
            const t = I18N[this.language];
            const div = document.createElement('div');
            div.className = 'message ' + msg.sender_type;
            let label = '';
            if (msg.sender_type === 'system') {
                label = msg.is_auto_reply ? t.auto_reply_label : t.system_label;
            } else if (msg.sender_type === 'manager') {
                label = t.manager_label;
            }
            const time = msg.created_at
                ? new Date(msg.created_at).toLocaleTimeString(this.language === 'ko' ? 'ko-KR' : 'en-US', { hour: '2-digit', minute: '2-digit' })
                : '';
            div.innerHTML = (label ? '<div class="message-label">' + label + '</div>' : '')
                + '<div class="message-bubble">' + this._escapeHtml(msg.content) + '</div>'
                + '<div class="message-time">' + time + '</div>';
            this.elements.messages.appendChild(div);
            this._scrollToBottom();
        }

        _requestManager() {
            if (this.managerStatus !== 'none' || !this.roomId) return;
            this.socket.emit('guest:request_manager', { roomId: this.roomId });
            this._updateManagerStatus('requested');
        }

        _updateManagerStatus(status) {
            this.managerStatus = status;
            const t = I18N[this.language];
            const btn = this.elements.managerBtn;
            btn.className = 'manager-connect-btn';
            if (status === 'requested') { btn.textContent = t.manager_requested; btn.classList.add('requested'); }
            else if (status === 'connected') { btn.textContent = t.manager_connected; btn.classList.add('connected'); }
            else { btn.textContent = t.manager_btn; }
        }

        _updateBadge() {
            const badge = this.elements.badge;
            if (this.unreadCount > 0) {
                badge.textContent = this.unreadCount > 9 ? '9+' : this.unreadCount;
                badge.classList.add('show');
            } else { badge.classList.remove('show'); }
        }

        _scrollToBottom() {
            setTimeout(() => { const el = this.elements.messages; el.scrollTop = el.scrollHeight; }, 50);
        }

        _escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
    }

    // ========== Auto-init ==========
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', () => new YakoreaChat());
    } else {
        new YakoreaChat();
    }
})();
